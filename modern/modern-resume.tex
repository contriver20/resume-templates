% Create education heading. Parameters: Name, degree, location, when
\newcommand{\educationHeading}[3]{\descript{#1}\hspace*{\fill}  \location{#3}\fakeNewLine}

% Create project heading. Parameters: Name, link, Tech stack
\newcommand{\projectHeading}[3]{\location{#1 | #2}\hfill
\location{#3}\\}

% create certification heading. Parameters: cert, date range
\newcommand{\certHeading}[2]{\location{#1 \hfill #2}\\}
